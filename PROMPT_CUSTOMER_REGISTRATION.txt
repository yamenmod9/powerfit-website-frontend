```dart
  final token = await _storage.read(key: 'access_token');
  return {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer $token',
  };
}

Future<Customer> createCustomer(CustomerRequest customer) async {
  final headers = await _getHeaders();
  final response = await http.post(
    Uri.parse('$baseUrl/api/customers'),  // ← Correct endpoint
    headers: headers,
    body: jsonEncode(customer.toJson()),
  );
  
  if (response.statusCode == 201) {
    final data = jsonDecode(response.body)['data'];
    return Customer.fromJson(data);
  } else if (response.statusCode == 404) {
    throw Exception('Endpoint not found - check API URL');
  } else if (response.statusCode == 401) {
    throw Exception('Not authenticated - user must login first');
  } else if (response.statusCode == 403) {
    throw Exception('Access denied - user role not authorized');
  } else {
    throw Exception('Failed to create customer: ${response.body}');
  }
}
```
